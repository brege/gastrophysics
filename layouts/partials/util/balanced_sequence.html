{{- $items := .items | default (slice) -}} {{- if gt (len $items) 0 -}} {{- $tag := .tag | default "div" -}} {{- $itemTag := cond (eq $tag "ul") "li" "span" -}} {{- $align := .align | default "center" -}} {{- $balanced := ne .balanced false -}} {{- $class := .class | default "" -}} {{- $sep := .separator | default "/" -}} {{- $baseClass := printf "balanced-sequence balance-%s" $align -}} {{- if $balanced }}{{- $baseClass = printf "%s balanced-wrap" $baseClass -}}{{- end -}} {{- $attrs := printf "class=\"%s%s\"" $baseClass (cond (ne $class "") (printf " %s" $class) "") -}} {{- with .id }}{{- $attrs = printf "%s id=\"%s\"" $attrs . -}}{{- end -}} {{- with .aria_label }}{{- $attrs = printf "%s aria-label=\"%s\"" $attrs . -}}{{- end -}} {{- printf "<%s %s>" $tag $attrs | safeHTML -}} {{- range $i, $item := $items }} {{- printf "<%s>" $itemTag | safeHTML -}} {{- if gt $i 0 -}} {{ $sep }} {{- end -}} {{- with $item.url -}} {{- $item.name -}} {{- else -}} {{ $item.name }} {{- end -}} {{- printf "" $itemTag | safeHTML -}} {{- end }} {{- printf "" $tag | safeHTML -}} {{- end -}}
